# Dans la vue

    {% for article in articles %}
        <article>
            <h2>{{article.title}}</h2>
        </article>
    {% endfor %}

Cette méthode appelle des getters sous le capot.

# Acceder à des fonctions

    {% for article in articles %}
        <article>
            <h2>{{article.getTitle()}}</h2>
        </article>
    {% endfor %}
